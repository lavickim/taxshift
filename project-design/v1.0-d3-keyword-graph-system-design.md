## 📊 데이터베이스 스키마 설계

### 1. 키워드 테이블 (`industry_keywords`)
```sql
CREATE TABLE industry_keywords (
    id BIGSERIAL PRIMARY KEY,
    keyword VARCHAR(100) NOT NULL,
    category VARCHAR(50), -- '제조업', 'IT', '서비스업' 등
    frequency INTEGER DEFAULT 1,
    confidence_score DECIMAL(3,2), -- 0.00-1.00
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW(),
    CONSTRAINT unique_keyword UNIQUE (keyword)
);

CREATE INDEX idx_keywords_category ON industry_keywords(category);
CREATE INDEX idx_keywords_frequency ON industry_keywords(frequency DESC);
```

### 2. 키워드 관계 테이블 (`keyword_relationships`)
```sql
CREATE TABLE keyword_relationships (
    id BIGSERIAL PRIMARY KEY,
    source_keyword_id BIGINT REFERENCES industry_keywords(id),
    target_keyword_id BIGINT REFERENCES industry_keywords(id),
    relationship_type VARCHAR(50), -- 'related', 'parent', 'child', 'synonym'
    strength DECIMAL(3,2), -- 관계 강도 0.00-1.00
    co_occurrence_count INTEGER DEFAULT 1,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    CONSTRAINT unique_relationship UNIQUE (source_keyword_id, target_keyword_id, relationship_type)
);

CREATE INDEX idx_relationships_source ON keyword_relationships(source_keyword_id);
CREATE INDEX idx_relationships_strength ON keyword_relationships(strength DESC);
```

### 3. 사업장 키워드 매핑 테이블 (`workplace_keywords`)
```sql
CREATE TABLE workplace_keywords (
    id BIGSERIAL PRIMARY KEY,
    workplace_name VARCHAR(255),
    industry_name VARCHAR(255),
    keyword_id BIGINT REFERENCES industry_keywords(id),
    extraction_method VARCHAR(50), -- 'nlp', 'manual', 'ai_generated'
    confidence DECIMAL(3,2),
    created_at TIMESTAMPTZ DEFAULT NOW(),
    FOREIGN KEY (workplace_name, industry_name) 
        REFERENCES datacollection_national_pension(workplace_name, industry_name)
);

CREATE INDEX idx_workplace_keywords_name ON workplace_keywords(workplace_name);
CREATE INDEX idx_workplace_keywords_keyword ON workplace_keywords(keyword_id);
```

## 🤖 키워드 추출 알고리즘 설계

### 1. NLP 파이프라인 (Python/Java)
```python
class IndustryKeywordExtractor:
    def __init__(self):
        self.korean_tokenizer = Okt()  # KoNLPy
        self.industry_dict = self._load_industry_dictionary()
        self.stop_words = {'업', '및', '기타', '관련', '서비스'}
    
    def extract_keywords(self, industry_name: str) -> List[KeywordResult]:
        """
        업종명에서 의미있는 키워드 추출
        
        Example:
        "컴퓨터 제조업" -> ["컴퓨터", "제조업", "IT장비", "하드웨어"]
        """
        # 1. 형태소 분석
        morphs = self.korean_tokenizer.morphs(industry_name)
        
        # 2. 불용어 제거
        filtered_morphs = [m for m in morphs if m not in self.stop_words]
        
        # 3. 산업 분류 사전 매칭
        industry_keywords = self._match_industry_dict(filtered_morphs)
        
        # 4. 복합어 처리 ("자동차제조" -> ["자동차", "제조"])
        compound_keywords = self._process_compounds(filtered_morphs)
        
        # 5. 신뢰도 점수 계산
        scored_keywords = self._calculate_confidence(
            industry_keywords + compound_keywords
        )
        
        return scored_keywords
    
    def _calculate_confidence(self, keywords: List[str]) -> List[KeywordResult]:
        """키워드 신뢰도 점수 계산"""
        results = []
        for keyword in keywords:
            confidence = 0.8  # 기본값
            
            # 산업 표준 용어면 +0.1
            if keyword in self.industry_dict:
                confidence += 0.1
                
            # 길이가 2자 이상이면 +0.05
            if len(keyword) >= 2:
                confidence += 0.05
                
            # 숫자나 특수문자 포함시 -0.2
            if any(c.isdigit() or not c.isalnum() for c in keyword):
                confidence -= 0.2
            
            results.append(KeywordResult(
                keyword=keyword,
                confidence=min(confidence, 1.0),
                category=self._categorize_keyword(keyword)
            ))
        
        return results
```

### 2. 관계 분석 알고리즘
```python
class KeywordRelationshipAnalyzer:
    def analyze_co_occurrence(self) -> List[Relationship]:
        """
        키워드 동시 출현 분석
        같은 업종에서 함께 나타나는 키워드들의 관계 강도 계산
        """
        # 1. 동시 출현 행렬 생성
        co_occurrence_matrix = self._build_co_occurrence_matrix()
        
        # 2. PMI (Pointwise Mutual Information) 계산
        pmi_scores = self._calculate_pmi(co_occurrence_matrix)
        
        # 3. 관계 유형 분류
        relationships = []
        for (keyword1, keyword2), pmi_score in pmi_scores.items():
            rel_type = self._classify_relationship(keyword1, keyword2, pmi_score)
            relationships.append(Relationship(
                source=keyword1,
                target=keyword2,
                type=rel_type,
                strength=min(pmi_score / 10.0, 1.0)  # 정규화
            ))
        
        return relationships
    
    def _classify_relationship(self, kw1: str, kw2: str, pmi: float) -> str:
        """관계 유형 분류"""
        # 제조업 관련 키워드들
        manufacturing_terms = {'제조업', '제조', '생산', '공장'}
        
        if kw1 in manufacturing_terms or kw2 in manufacturing_terms:
            return 'manufacturing_related'
        elif self._is_parent_child(kw1, kw2):
            return 'hierarchical'
        elif pmi > 5.0:
            return 'strongly_related'
        else:
            return 'related'
```

### 2. D3 Force Layout 구현
```typescript
// components/d3-keyword-graph.tsx
export function KeywordNetworkGraph({ data, width = 800, height = 600 }: KeywordGraphProps) {
    const svgRef = useRef<SVGSVGElement>(null);
    
    useEffect(() => {
        if (!data || !svgRef.current) return;
        
        const svg = d3.select(svgRef.current);
        svg.selectAll("*").remove(); // 기존 요소 제거
        
        // 1. 색상 스케일 설정
        const colorScale = d3.scaleOrdinal()
            .domain(data.categories)
            .range(['#FF6B6B', '#4ECDC4', '#45B7D1', '#96CEB4', '#FFEAA7', '#DDA0DD']);
        
        // 2. 노드 크기 스케일
        const radiusScale = d3.scaleSqrt()
            .domain(d3.extent(data.nodes, d => d.frequency))
            .range([5, 30]);
        
        // 3. Force Simulation 설정
        const simulation = d3.forceSimulation(data.nodes)
            .force("link", d3.forceLink(data.links)
                .id(d => d.id)
                .distance(d => 50 + (1 - d.strength) * 100)
                .strength(d => d.strength)
            )
            .force("charge", d3.forceManyBody()
                .strength(-300)
                .distanceMax(200)
            )
            .force("center", d3.forceCenter(width / 2, height / 2))
            .force("collision", d3.forceCollide()
                .radius(d => radiusScale(d.frequency) + 2)
            );
        
        // 4. 링크 그리기
        const links = svg.append("g")
            .selectAll("line")
            .data(data.links)
            .enter().append("line")
            .attr("stroke", "#999")
            .attr("stroke-opacity", 0.6)
            .attr("stroke-width", d => Math.sqrt(d.strength * 10));
        
        // 5. 노드 그리기
        const nodes = svg.append("g")
            .selectAll("circle")
            .data(data.nodes)
            .enter().append("circle")
            .attr("r", d => radiusScale(d.frequency))
            .attr("fill", d => colorScale(d.category))
            .attr("stroke", "#fff")
            .attr("stroke-width", 2)
            .call(d3.drag()
                .on("start", dragstarted)
                .on("drag", dragged)
                .on("end", dragended)
            );
        
        // 6. 라벨 추가
        const labels = svg.append("g")
            .selectAll("text")
            .data(data.nodes)
            .enter().append("text")
            .text(d => d.keyword)
            .attr("font-size", 12)
            .attr("text-anchor", "middle")
            .attr("dy", ".35em")
            .style("pointer-events", "none");
        
        // 7. 툴팁 기능
        nodes
            .on("mouseover", (event, d) => {
                // 툴팁 표시 로직
                showTooltip(event, d);
            })
            .on("mouseout", hideTooltip);
        
        // 8. 시뮬레이션 업데이트
        simulation.on("tick", () => {
            links
                .attr("x1", d => d.source.x)
                .attr("y1", d => d.source.y)
                .attr("x2", d => d.target.x)
                .attr("y2", d => d.target.y);
            
            nodes
                .attr("cx", d => d.x)
                .attr("cy", d => d.y);
            
            labels
                .attr("x", d => d.x)
                .attr("y", d => d.y);
        });
        
        // Drag 함수들
        function dragstarted(event, d) {
            if (!event.active) simulation.alphaTarget(0.3).restart();
            d.fx = d.x;
            d.fy = d.y;
        }
        
        function dragged(event, d) {
            d.fx = event.x;
            d.fy = event.y;
        }
        
        function dragended(event, d) {
            if (!event.active) simulation.alphaTarget(0);
            d.fx = null;
            d.fy = null;
        }
        
    }, [data, width, height]);
    
    return (
        <div className="keyword-network-container">
            <svg ref={svgRef} width={width} height={height} />
        </div>
    );
}
```
