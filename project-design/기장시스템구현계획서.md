# MoneyShift AI 기장시스템 구현계획서

## 📋 프로젝트 개요

### 현재 상황
- ✅ **4-Layer Processing Pipeline** 완전 구축 (캐시, 키워드, 브랜드, LLM)
- ✅ **80개 키워드 그룹 + 40+ 태그 시스템** 운영 중
- ✅ **Java API + NextJS 어드민 대시보드** 완전 연동
- ✅ **tag_account_mappings** (태그→계정과목 자동 매핑) 구현
- ✅ **89% 분류 성공률** 달성

### 구현 목표
기존 룰엔진 시스템과 연계하여 **복식부기 기반 자동 기장 시스템** 구축
- 거래 분류 → 분개 생성 → 재무제표 생성의 완전 자동화
- 어드민 툴에서 테스트 및 관리 가능한 통합 시스템

```
[기존 룰엔진 시스템]
```

#### 3.2 기장 처리 엔진 서비스
```java
@Service
@Transactional
public class AccountingEngine {
    
    @Autowired
    private JournalEntryRepository journalEntryRepository;
    
    @Autowired
    private ChartOfAccountsRepository chartOfAccountsRepository;
    
    @Autowired
    private TransactionRepository transactionRepository;
    
    /**
     * 거래 내역을 분석하여 자동으로 분개 생성
     */
    public JournalEntry processTransaction(TransactionToJournalRequest request) {
        
        // 1. 기존 거래 분류 시스템 활용 (키워드 → 태그 → 계정과목)
        Transaction transaction = getTransaction(request.getTransactionId());
        String suggestedTag = transaction.getFinalSuggestedTag();
        String accountCode = getAccountCodeFromTag(suggestedTag);
        
        // 2. 복식부기 분개 생성
        JournalEntry journalEntry = new JournalEntry();
        journalEntry.setCompanyId(transaction.getCompanyId());
        journalEntry.setEntryDate(transaction.getTransactionDate());
        journalEntry.setDescription(transaction.getRawText());
        journalEntry.setReferenceType("TRANSACTION");
        journalEntry.setReferenceId(transaction.getId());
        journalEntry.setTotalAmount(transaction.getAmount());
        
        // 3. 차변/대변 계산
        List<JournalEntryDetail> details = generateDoubleEntry(
            accountCode, 
            transaction.getAmount(), 
            transaction.getTransactionType()
        );
        
        journalEntry.setDetails(details);
        
        // 4. 분개 검증 및 저장
        validateJournalEntry(journalEntry);
        return journalEntryRepository.save(journalEntry);
    }
    
    /**
     * 복식부기 원리에 따른 차변/대변 자동 생성
     */
    private List<JournalEntryDetail> generateDoubleEntry(
        String expenseAccountCode, 
        Long amount, 
        TransactionType transactionType) {
        
        List<JournalEntryDetail> details = new ArrayList<>();
        
        if (transactionType == TransactionType.EXPENSE) {
            // 비용 발생: 차변(비용 계정), 대변(현금/예금)
            details.add(createJournalDetail(1, expenseAccountCode, amount, 0L, "비용 발생"));
            details.add(createJournalDetail(2, "1120", 0L, amount, "현금 지출")); // 보통예금
            
        } else if (transactionType == TransactionType.INCOME) {
            // 수익 발생: 차변(현금/예금), 대변(수익 계정)
            details.add(createJournalDetail(1, "1120", amount, 0L, "현금 수입")); // 보통예금
            details.add(createJournalDetail(2, "4110", 0L, amount, "매출 발생")); // 매출
        }
        
        return details;
    }
    
    /**
     * 재무제표 생성
     */
    public FinancialStatement generateFinancialStatement(
        String companyId, 
        String statementType, 
        LocalDate periodStart, 
        LocalDate periodEnd) {
        
        switch (statementType) {
            case "대차대조표":
                return generateBalanceSheet(companyId, periodEnd);
            case "손익계산서":
                return generateIncomeStatement(companyId, periodStart, periodEnd);
            default:
                throw new IllegalArgumentException("지원하지 않는 재무제표 유형: " + statementType);
        }
    }
    
    /**
     * 대차대조표 생성
     */
    private FinancialStatement generateBalanceSheet(String companyId, LocalDate asOfDate) {
        
        Map<String, Object> balanceSheet = new HashMap<>();
        
        // 자산 계정 집계
        Map<String, Long> assets = aggregateAccountsByType(companyId, asOfDate, "자산");
        Map<String, Long> currentAssets = filterBySubtype(assets, "유동자산");
        Map<String, Long> nonCurrentAssets = filterBySubtype(assets, "비유동자산");
        
        // 부채 계정 집계
        Map<String, Long> liabilities = aggregateAccountsByType(companyId, asOfDate, "부채");
        Map<String, Long> currentLiabilities = filterBySubtype(liabilities, "유동부채");
        Map<String, Long> nonCurrentLiabilities = filterBySubtype(liabilities, "비유동부채");
        
        // 자본 계정 집계
        Map<String, Long> equity = aggregateAccountsByType(companyId, asOfDate, "자본");
        
        // 대차대조표 구조 생성
        balanceSheet.put("자산", Map.of(
            "유동자산", currentAssets,
            "비유동자산", nonCurrentAssets,
            "자산총계", sumValues(assets)
        ));
        
        balanceSheet.put("부채", Map.of(
            "유동부채", currentLiabilities,
            "비유동부채", nonCurrentLiabilities,
            "부채총계", sumValues(liabilities)
        ));
        
        balanceSheet.put("자본", Map.of(
            "자본금", equity,
            "자본총계", sumValues(equity)
        ));
        
        // 대차 균형 검증
        Long totalAssets = sumValues(assets);
        Long totalLiabilitiesAndEquity = sumValues(liabilities) + sumValues(equity);
        
        if (!totalAssets.equals(totalLiabilitiesAndEquity)) {
            throw new RuntimeException("대차대조표 불균형: 자산=" + totalAssets + ", 부채+자본=" + totalLiabilitiesAndEquity);
        }
        
        return new FinancialStatement("대차대조표", asOfDate, asOfDate, balanceSheet);
    }
    
    /**
     * 손익계산서 생성
     */
    private FinancialStatement generateIncomeStatement(String companyId, LocalDate periodStart, LocalDate periodEnd) {
        
        Map<String, Object> incomeStatement = new HashMap<>();
        
        // 수익 계정 집계
        Map<String, Long> revenues = aggregateAccountsByTypeAndPeriod(companyId, periodStart, periodEnd, "수익");
        
        // 비용 계정 집계
        Map<String, Long> expenses = aggregateAccountsByTypeAndPeriod(companyId, periodStart, periodEnd, "비용");
        
        // 손익계산서 구조 생성
        Long totalRevenue = sumValues(revenues);
        Long totalExpenses = sumValues(expenses);
        Long netIncome = totalRevenue - totalExpenses;
        
        incomeStatement.put("수익", Map.of(
            "매출액", revenues.getOrDefault("매출", 0L),
            "기타수익", revenues.getOrDefault("기타수익", 0L),
            "수익총계", totalRevenue
        ));
        
        incomeStatement.put("비용", Map.of(
            "판매관리비", filterBySubtype(expenses, "판매관리비"),
            "기타비용", filterBySubtype(expenses, "기타비용"),
            "비용총계", totalExpenses
        ));
        
        incomeStatement.put("손익", Map.of(
            "당기순이익", netIncome
        ));
        
        return new FinancialStatement("손익계산서", periodStart, periodEnd, incomeStatement);
    }
    
    /**
     * 분개 검증 (차변 = 대변)
     */
    private void validateJournalEntry(JournalEntry journalEntry) {
        Long totalDebit = journalEntry.getDetails().stream()
            .mapToLong(JournalEntryDetail::getDebitAmount)
            .sum();
            
        Long totalCredit = journalEntry.getDetails().stream()
            .mapToLong(JournalEntryDetail::getCreditAmount)
            .sum();
            
        if (!totalDebit.equals(totalCredit)) {
            throw new RuntimeException("분개 불균형: 차변=" + totalDebit + ", 대변=" + totalCredit);
        }
        
        if (!totalDebit.equals(journalEntry.getTotalAmount())) {
            throw new RuntimeException("분개 금액 불일치: 총액=" + journalEntry.getTotalAmount() + ", 차변합계=" + totalDebit);
        }
    }
}
```

```typescript
// components/accounting/transaction-to-journal-test.tsx
export function TransactionToJournalTest() {
  const [selectedTransaction, setSelectedTransaction] = useState<Transaction>();
  const [generatedJournal, setGeneratedJournal] = useState<JournalEntry>();
  const [isProcessing, setIsProcessing] = useState(false);

  const handleProcessTransaction = async () => {
    if (!selectedTransaction) return;
    
    setIsProcessing(true);
    try {
      const response = await fetch('/api/v2/accounting/process-transaction', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ transactionId: selectedTransaction.id })
      });
      
      const result = await response.json();
      setGeneratedJournal(result.journalEntry);
    } catch (error) {
      console.error('분개 생성 오류:', error);
    } finally {
      setIsProcessing(false);
    }
  };

  return (
    <div className="space-y-6">
      <Card>
        <CardHeader>
          <CardTitle className="flex items-center gap-2">
            <Calculator className="h-5 w-5" />
            거래 → 분개 자동 생성 테스트
          </CardTitle>
        </CardHeader>
        <CardContent className="space-y-4">
          {/* 거래 선택 */}
          <TransactionSelector 
            onTransactionSelect={setSelectedTransaction}
            selectedTransaction={selectedTransaction}
          />
          
          {/* 처리 버튼 */}
          <Button 
            onClick={handleProcessTransaction}
            disabled={!selectedTransaction || isProcessing}
            className="w-full"
          >
            {isProcessing ? '분개 생성 중...' : '분개 생성'}
```

```typescript
              <label className="text-sm font-medium">종료일</label>
              <Input 
                type="date" 
                value={dateRange.end}
                onChange={(e) => setDateRange(prev => ({ ...prev, end: e.target.value }))}
              />
            </div>
          </div>

          {/* 생성 버튼 */}
          <Button 
            onClick={handleGenerateStatement}
            disabled={!dateRange.start || !dateRange.end || isGenerating}
            className="w-full"
          >
            {isGenerating ? '생성 중...' : '재무제표 생성'}
          </Button>
        </CardContent>
      </Card>

      {/* 재무제표 미리보기 */}
      <Card>
        <CardHeader>
          <CardTitle>{statementType}</CardTitle>
        </CardHeader>
        <CardContent>
          {statement ? (
            <FinancialStatementPreview 
              statement={statement} 
              type={statementType}
            />
          ) : (
            <div className="text-center py-8 text-muted-foreground">
              설정을 입력하고 '재무제표 생성' 버튼을 클릭하세요.
            </div>
          )}
        </CardContent>
      </Card>
    </div>
  );
}
```
