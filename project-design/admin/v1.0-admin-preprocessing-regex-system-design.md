## 📋 시스템 개요

**어드민 전처리 정규식 기능**은 MoneyShift 어드민 툴의 핵심 기능으로, 수집된 거래 문자열을 바탕으로 **인텔리전트한 정규식 패턴 관리 시스템**을 제공합니다. 사용자가 직접 거래 문자열을 입력하고, LLM의 도움을 받아 최적의 정규식 패턴을 생성하고 관리할 수 있는 **차세대 전처리 시스템**입니다.

### 🎯 핵심 목적
1. **지인들로부터 수집한 거래 문자열**을 효율적으로 처리
2. **LLM 기반 인텔리전트 패턴 제안** 시스템
3. **충돌 감지 및 우선순위 자동 조정** 기능
4. **실시간 테스트 및 검증** 환경
5. **어드민 툴 완성 후 백엔드 적용** 단계적 접근

### 🏗️ 시스템 아키텍처
```
┌─────────────────────────────────────────────────────────────────┐
│                    어드민 전처리 정규식 기능                          │
├─────────────────────────────────────────────────────────────────┤
│ 1. 거래문자열 입력 → 2. LLM 패턴 제안 → 3. 충돌 분석 → 4. 우선순위 조정 │
│ 5. 실시간 테스트 → 6. 패턴 저장 → 7. 백엔드 동기화 (향후)            │
└─────────────────────────────────────────────────────────────────┘
```

### 3. ⚔️ **충돌 감지 및 분석 시스템**

#### 3.1 패턴 충돌 감지 알고리즘
```typescript
interface ConflictAnalysis {
  conflictType: 'OVERLAP' | 'PRECEDENCE' | 'AMBIGUITY' | 'NONE';
  severity: 'HIGH' | 'MEDIUM' | 'LOW';
  description: string;
  affectedPatterns: string[];
  suggestedResolution: string;
  testCaseConflicts: TestCaseConflict[];
}

interface TestCaseConflict {
  input: string;
  pattern1Result: string;
  pattern2Result: string;
  expectedResult?: string;
  recommendation: string;
}

class ConflictDetectionEngine {
  /**
   * 패턴 간 충돌 감지
   */
  detectConflicts(
    newPattern: RegexPattern, 
    existingPatterns: RegexPattern[]
  ): ConflictAnalysis[] {
    const conflicts: ConflictAnalysis[] = [];
    
    for (const existing of existingPatterns) {
      // 1. 정규식 오버랩 검사
      const overlapConflict = this.checkRegexOverlap(newPattern, existing);
      if (overlapConflict) conflicts.push(overlapConflict);
      
      // 2. 우선순위 충돌 검사
      const precedenceConflict = this.checkPrecedenceConflict(newPattern, existing);
      if (precedenceConflict) conflicts.push(precedenceConflict);
      
      // 3. 테스트 케이스 기반 충돌 검사
      const testConflicts = this.checkTestCaseConflicts(newPattern, existing);
      conflicts.push(...testConflicts);
    }
    
    return conflicts;
  }
  
  /**
   * 정규식 패턴 오버랩 검사
   */
  private checkRegexOverlap(pattern1: RegexPattern, pattern2: RegexPattern): ConflictAnalysis | null {
    // 샘플 문자열들로 교차 테스트
    const testStrings = this.generateTestStrings();
    const conflicts: TestCaseConflict[] = [];
    
    for (const testString of testStrings) {
      const result1 = this.applyPattern(pattern1, testString);
      const result2 = this.applyPattern(pattern2, testString);
      
      if (result1.matched && result2.matched && result1.output !== result2.output) {
        conflicts.push({
          input: testString,
          pattern1Result: result1.output,
          pattern2Result: result2.output,
          recommendation: "우선순위 조정 필요"
        });
      }
    }
    
    return conflicts.length > 0 ? {
      conflictType: 'OVERLAP',
      severity: conflicts.length > 5 ? 'HIGH' : conflicts.length > 2 ? 'MEDIUM' : 'LOW',
      description: `${conflicts.length}개의 테스트 케이스에서 서로 다른 결과 생성`,
      affectedPatterns: [pattern1.name, pattern2.name],
      suggestedResolution: "더 구체적인 패턴을 높은 우선순위로 배치",
      testCaseConflicts: conflicts
    } : null;
  }
}
```

### 4. 🎛️ **우선순위 관리 시스템**

#### 4.1 인텔리전트 우선순위 제안
```typescript
interface PriorityRecommendation {
  patternId: string;
  currentPriority: number;
  suggestedPriority: number;
  reasoning: string;
  impact: {
    affectedPatterns: string[];
    expectedImprovements: string[];
    potentialRisks: string[];
  };
  confidence: number;  // LLM 제안 신뢰도
}

class IntelligentPriorityManager {
  /**
   * 전체 패턴의 최적 우선순위 제안
   */
  async suggestOptimalPriorities(
    patterns: RegexPattern[],
    testTransactions: string[]
  ): Promise<PriorityRecommendation[]> {
    
    // 1. 패턴 특성 분석
    const patternCharacteristics = patterns.map(p => ({
      id: p.name,
      specificity: this.calculateSpecificity(p),
      coverage: this.calculateCoverage(p, testTransactions),
      businessImportance: this.assessBusinessImportance(p),
      performance: this.estimatePerformance(p)
    }));
    
    // 2. LLM 기반 우선순위 제안 요청
    const llmSuggestion = await this.requestLLMPrioritySuggestion(
      patterns, 
      testTransactions,
      patternCharacteristics
    );
    
    // 3. 알고리즘 기반 검증
    const validatedSuggestions = this.validateSuggestions(llmSuggestion, patternCharacteristics);
    
    return validatedSuggestions;
  }
  
  /**
   * 패턴 구체성 계산 (더 구체적일수록 높은 우선순위)
   */
  private calculateSpecificity(pattern: RegexPattern): number {
    let score = 0;
    
    // 정규식 복잡도
    score += pattern.inputRegex.length * 0.1;
    
    // 특수 문자 사용 정도
    const specialChars = pattern.inputRegex.match(/[\[\](){}.*+?^$|\\]/g);
    score += (specialChars?.length || 0) * 2;
    
    // 고정 문자열 길이
    const fixedStrings = pattern.inputRegex.match(/[가-힣a-zA-Z0-9]+/g);
    const totalFixedLength = fixedStrings?.reduce((sum, str) => sum + str.length, 0) || 0;
    score += totalFixedLength * 0.5;
    
    return score;
  }
}
```

```typescript
interface TestingWorkspace {
  patterns: RegexPattern[];
  testTransactions: string[];
  results: TestResult[];
  insights: TestInsight[];
}

interface TestResult {
  input: string;
  appliedPattern?: RegexPattern;
  output: string;
  processingTime: number;
  isCorrect?: boolean;  // 사용자 피드백
  alternatives: AlternativeResult[];
}

interface TestInsight {
  type: 'ACCURACY' | 'PERFORMANCE' | 'COVERAGE' | 'CONFLICT';
  message: string;
  severity: 'INFO' | 'WARNING' | 'ERROR';
  suggestedAction?: string;
}

// React 컴포넌트 예시
const RegexTestingWorkspace: React.FC = () => {
  const [workspace, setWorkspace] = useState<TestingWorkspace>({
    patterns: [],
    testTransactions: [],
    results: [],
    insights: []
  });
  
  // 실시간 테스트 실행
  const runLiveTest = async (input: string) => {
    const results = await Promise.all(
      workspace.patterns.map(async (pattern) => {
        const startTime = performance.now();
        const result = applyPattern(pattern, input);
        const endTime = performance.now();
        
        return {
          pattern,
          result: result.output,
          matched: result.matched,
          processingTime: endTime - startTime
        };
      })
    );
    
    // 결과 우선순위 정렬
    const sortedResults = results
      .filter(r => r.matched)
      .sort((a, b) => b.pattern.priority - a.pattern.priority);
    
    const finalResult: TestResult = {
      input,
      appliedPattern: sortedResults[0]?.pattern,
      output: sortedResults[0]?.result || input,
      processingTime: sortedResults.reduce((sum, r) => sum + r.processingTime, 0),
      alternatives: sortedResults.slice(1).map(r => ({
        pattern: r.pattern,
        output: r.result,
        processingTime: r.processingTime
      }))
    };
    
    setWorkspace(prev => ({
      ...prev,
      results: [...prev.results, finalResult]
    }));
    
    // 인사이트 생성
    const insights = generateTestInsights(finalResult, sortedResults);
    setWorkspace(prev => ({
      ...prev,
      insights: [...prev.insights, ...insights]
    }));
  };
  
  return (
    <div className="testing-workspace">
      {/* 테스트 인터페이스 UI */}
    </div>
  );
};
```

```typescript
class RealtimePatternTester {
  private patterns: RegexPattern[] = [];
  private testCache: Map<string, TestResult> = new Map();
  
  /**
   * 실시간 테스트 실행
   */
  async testTransaction(input: string): Promise<TestResult> {
    const cacheKey = `${input}_${this.getPatternsHash()}`;
    
    if (this.testCache.has(cacheKey)) {
      return this.testCache.get(cacheKey)!;
    }
    
    const startTime = performance.now();
    const results: PatternTestResult[] = [];
    
    // 우선순위 순으로 패턴 적용
    const sortedPatterns = [...this.patterns].sort((a, b) => b.priority - a.priority);
    
    for (const pattern of sortedPatterns) {
      const patternStartTime = performance.now();
      
      try {
        const regex = new RegExp(pattern.inputRegex, 'gi');
        const match = regex.exec(input);
        
        if (match) {
          const output = this.applyOutputTemplate(pattern.outputTemplate, match);
          const patternEndTime = performance.now();
          
          results.push({
            pattern,
            matched: true,
            output,
            processingTime: patternEndTime - patternStartTime,
            matchGroups: match.slice(1)
          });
          
          // 첫 번째 매치로 결정 (우선순위가 높은 패턴)
          break;
        }
      } catch (error) {
        results.push({
          pattern,
          matched: false,
          error: error.message,
          processingTime: performance.now() - patternStartTime
        });
      }
    }
    
    const totalTime = performance.now() - startTime;
    const finalResult: TestResult = {
      input,
      appliedPattern: results.find(r => r.matched)?.pattern,
      output: results.find(r => r.matched)?.output || input,
      processingTime: totalTime,
      allResults: results,
      timestamp: new Date()
    };
    
    this.testCache.set(cacheKey, finalResult);
    return finalResult;
  }
  
  /**
   * 배치 테스트 실행
   */
  async testBatch(inputs: string[]): Promise<BatchTestResult> {
    const results = await Promise.all(
      inputs.map(input => this.testTransaction(input))
    );
    
    const stats = this.calculateBatchStats(results);
    
    return {
      results,
      totalInputs: inputs.length,
      successfulMatches: results.filter(r => r.appliedPattern).length,
      averageProcessingTime: stats.averageTime,
      accuracy: stats.accuracy,
      insights: this.generateBatchInsights(results)
    };
  }
  
  /**
   * 출력 템플릿 적용
   */
  private applyOutputTemplate(template: string, match: RegExpExecArray): string {
    let output = template;
    
    // $1, $2, ... 치환
    for (let i = 1; i < match.length; i++) {
      output = output.replace(new RegExp(`\\$${i}`, 'g'), match[i] || '');
    }
    
    // 특수 치환 함수들
    output = output.replace(/\$TRIM\((.+?)\)/g, (_, content) => content.trim());
    output = output.replace(/\$UPPER\((.+?)\)/g, (_, content) => content.toUpperCase());
    output = output.replace(/\$LOWER\((.+?)\)/g, (_, content) => content.toLowerCase());
    
    return output.trim();
  }
}
```

```typescript
class PatternValidationSystem {
  /**
   * 정규식 패턴 검증
   */
  validatePattern(pattern: RegexPattern): ValidationResult {
    const errors: string[] = [];
    const warnings: string[] = [];
    
    // 1. 정규식 문법 검증
    try {
      new RegExp(pattern.inputRegex);
    } catch (error) {
      errors.push(`잘못된 정규식 문법: ${error.message}`);
    }
    
    // 2. 성능 검증 (복잡도 체크)
    const complexity = this.calculateRegexComplexity(pattern.inputRegex);
    if (complexity > 100) {
      warnings.push(`정규식이 너무 복잡합니다 (복잡도: ${complexity}). 성능 문제가 발생할 수 있습니다.`);
    }
    
    // 3. 보안 검증 (ReDoS 취약점 체크)
    const redosRisk = this.checkReDoSVulnerability(pattern.inputRegex);
    if (redosRisk.isVulnerable) {
      errors.push(`ReDoS 취약점 발견: ${redosRisk.description}`);
    }
    
    // 4. 출력 템플릿 검증
    const templateErrors = this.validateOutputTemplate(pattern.outputTemplate, pattern.inputRegex);
    errors.push(...templateErrors);
    
    // 5. 테스트 케이스 검증
    if (pattern.testCases && pattern.testCases.length > 0) {
      for (const testCase of pattern.testCases) {
        const testResult = this.testPatternAgainstCase(pattern, testCase);
        if (!testResult.passed) {
          warnings.push(`테스트 케이스 실패: "${testCase.input}" -> 예상: "${testCase.expected}", 실제: "${testResult.actual}"`);
        }
      }
    }
    
    return {
      isValid: errors.length === 0,
      errors,
      warnings,
      complexity,
      securityRisk: redosRisk.isVulnerable ? 'HIGH' : 'LOW'
    };
  }
  
  /**
   * ReDoS (Regular Expression Denial of Service) 취약점 검사
   */
  private checkReDoSVulnerability(regex: string): { isVulnerable: boolean; description: string } {
    // 위험한 패턴들 검사
    const dangerousPatterns = [
      /(\.\*|\.\+)\?.*(\.\*|\.\+)/,  // 중첩된 quantifier
      /\(([^)]*\+[^)]*)+\)/,         // 중첩된 그룹 내 quantifier
      /\([^)]*\*[^)]*\*[^)]*\)/,     // 여러 * quantifier
    ];
    
    for (const pattern of dangerousPatterns) {
      if (pattern.test(regex)) {
        return {
          isVulnerable: true,
          description: '중첩된 quantifier로 인한 백트래킹 위험'
        };
      }
    }
    
    return { isVulnerable: false, description: '' };
  }
}
```

```typescript
class PerformanceOptimizer {
  private patternCache: Map<string, CompiledPattern> = new Map();
  private resultCache: Map<string, TestResult> = new Map();
  
  /**
   * 정규식 컴파일 캐싱
   */
  getCompiledPattern(pattern: RegexPattern): CompiledPattern {
    const cacheKey = pattern.name;
    
    if (!this.patternCache.has(cacheKey)) {
      const compiled = {
        regex: new RegExp(pattern.inputRegex, 'gi'),
        template: pattern.outputTemplate,
        priority: pattern.priority,
        compiledAt: Date.now()
      };
      this.patternCache.set(cacheKey, compiled);
    }
    
    return this.patternCache.get(cacheKey)!;
  }
  
  /**
   * 결과 캐싱 (Redis 연동)
   */
  async getCachedResult(input: string, patternsHash: string): Promise<TestResult | null> {
    const cacheKey = `regex_test:${input}:${patternsHash}`;
    
    // 메모리 캐시 먼저 확인
    if (this.resultCache.has(cacheKey)) {
      return this.resultCache.get(cacheKey)!;
    }
    
    // Redis 캐시 확인
    const cached = await redisClient.get(cacheKey);
    if (cached) {
      const result = JSON.parse(cached);
      this.resultCache.set(cacheKey, result);
      return result;
    }
    
    return null;
  }
  
  async setCachedResult(input: string, patternsHash: string, result: TestResult): Promise<void> {
    const cacheKey = `regex_test:${input}:${patternsHash}`;
    
    // 메모리 캐시
    this.resultCache.set(cacheKey, result);
    
    // Redis 캐시 (1시간 TTL)
    await redisClient.setex(cacheKey, 3600, JSON.stringify(result));
  }
```

```typescript
    if (this.containsXSSPatterns(input)) {
      issues.push({
        type: 'XSS',
        severity: 'HIGH',
        description: 'XSS 공격 패턴 감지'
      });
    }
    
    // SQL Injection 방지
    if (this.containsSQLInjectionPatterns(input)) {
      issues.push({
        type: 'SQL_INJECTION', 
        severity: 'HIGH',
        description: 'SQL 인젝션 패턴 감지'
      });
    }
    
    // 정규식 길이 제한
    if (input.length > 1000) {
      issues.push({
        type: 'LENGTH_LIMIT',
        severity: 'MEDIUM', 
        description: '입력 길이가 제한을 초과함'
      });
    }
    
    return {
      isValid: issues.filter(i => i.severity === 'HIGH').length === 0,
      issues
    };
  }
  
  /**
   * LLM API 요청 보안
   */
  sanitizeLLMRequest(request: any): any {
    return {
      ...request,
      // 민감한 정보 마스킹
      rawTransactions: request.rawTransactions?.map(t => this.maskSensitiveInfo(t)),
      userNotes: this.sanitizeUserNotes(request.userNotes)
    };
  }
  
  private maskSensitiveInfo(text: string): string {
    // 카드번호, 계좌번호 등 마스킹
    return text
      .replace(/\d{4}-\d{4}-\d{4}-\d{4}/g, '****-****-****-****')
      .replace(/\d{10,}/g, (match) => '*'.repeat(match.length));
  }
}
```
